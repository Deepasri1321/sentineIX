buttons.forEach(btn=>{
btn.classList.remove('active-btn');
});

}

/* AI PREDICTION */

async function predictManipulation(){

const stock=document.getElementById('stockSelect').value;

const output=document.getElementById('predictionOutput');

output.innerHTML=`
<div style="color:#00e5ff;font-size:20px;">
Analyzing ${stock} with SentinelX AI...
</div>
`;

await new Promise(resolve=>setTimeout(resolve,1500));

const response = await fetch(
  `https://www.alphavantage.co/query?function=TIME_SERIES_DAILY&symbol=${stock}&apikey=WNBT1HIZUR28GR6V`
);

const data = await response.json();

const series = data["Time Series (Daily)"];

const days = Object.keys(series).slice(0, 2);

const todayClose = parseFloat(series[days[0]]["4. close"]);
const yesterdayClose = parseFloat(series[days[1]]["4. close"]);

const change = ((todayClose - yesterdayClose) / yesterdayClose) * 100;

const risk = Math.min(Math.abs(Math.round(change * 10)), 100);

const confidence=Math.floor(80+Math.random()*20);

let status='';
let color='';

if(risk>=75){
status='HIGH RISK';
color='#ff3b3b';
}
else if(risk>=45){
status='MEDIUM RISK';
color='#ffbf00';
}
else{
status='LOW RISK';
color='#00ff9d';
}

output.innerHTML=`

<h2 style="
color:${color};
font-size:30px;
margin-bottom:16px;
">
${status}
</h2>

<p style="margin-bottom:10px;">
📈 Stock: <b>${stock}</b>
</p>

<p style="margin-bottom:10px;">
⚠ Manipulation Probability: <b>${risk}%</b>
</p>

<p style="margin-bottom:10px;">
🤖 AI Confidence: <b>${confidence}%</b>
</p>

<p>
🔍 Suspicious pattern activity detected by predictive engine.
</p>

`;

document.getElementById('riskValue').innerText=risk+'%';

}

async function loadStockHistory(symbol = 'TSLA') {
  const API_KEY = 'WNBT1HIZUR28GR6V';

  try {
    const response = await fetch(
      `https://www.alphavantage.co/query?function=TIME_SERIES_DAILY&symbol=${symbol}&apikey=${API_KEY}`
    );

    const data = await response.json();

    const series = data["Time Series (Daily)"];

    if (!series) {
      console.log(data);
      alert("Stock data not found");
      return;
    }

    const dates = Object.keys(series)
      .slice(0, 10)
      .reverse();

    const prices = dates.map(date =>
      parseFloat(series[date]["4. close"])
    );

    marketChart.data.labels = dates;
    marketChart.data.datasets[0].data = prices;
    marketChart.update();
    const table = document.getElementById('stockHistoryTable');

table.innerHTML = '';

dates.forEach((date, index) => {
  table.innerHTML += `
    <tr>
      <td>${date}</td>
      <td>₹${(prices[index] * 83).toFixed(2)}</td>
    </tr>
  `;
});

  } catch (error) {
    console.error(error);
    alert("API request failed");
  }
}
